    def close_window(self, window_name):
        """Schließt ein Fenster sicher und entfernt die Referenz"""
        if self.active_windows[window_name] and self.active_windows[window_name].winfo_exists():
            self.active_windows[window_name].destroy()
        self.active_windows[window_name] = None

    def ensure_single_window(self, window_name):
        """Stellt sicher, dass nur ein Fenster des angegebenen Typs existiert"""
        if self.active_windows[window_name] and self.active_windows[window_name].winfo_exists():
            self.active_windows[window_name].lift()
            self.active_windows[window_name].focus_force()
            return False
        return True

    def create_camera_lock(self):
        """Erstellt eine Lock-Datei für die Kamera."""
        try:
            with open(self.camera_lock_file, 'w') as f:
                f.write(f"TuxGuard Camera Lock - PID: {os.getpid()}\n")
                f.write(f"Timestamp: {time.time()}\n")
            print(f"Kamera-Lock erstellt: {self.camera_lock_file}")
            return True
        except Exception as e:
            print(f"Fehler beim Erstellen der Kamera-Lock: {e}")
            return False

    def remove_camera_lock(self):
        """Entfernt die Kamera-Lock-Datei."""
        try:
            if os.path.exists(self.camera_lock_file):
                os.remove(self.camera_lock_file)
                print(f"Kamera-Lock entfernt: {self.camera_lock_file}")
        except Exception as e:
            print(f"Fehler beim Entfernen der Kamera-Lock: {e}")

    def check_camera_access(self):
        """Überprüft, ob andere Programme auf die Kamera zugreifen wollen."""
        try:
            # Prüfe, ob andere Programme die Kamera öffnen wollen
            if os.path.exists(self.camera_device):
                # Versuche die Kamera zu öffnen
                fd = os.open(self.camera_device, os.O_RDWR | os.O_NONBLOCK)
                os.close(fd)
                return True
        except (OSError, IOError) as e:
            if e.errno == 16:  # EBUSY - Device or resource busy
                print(f"Kamera wird von anderem Programm verwendet: {e}")
                return False
            else:
                print(f"Fehler beim Prüfen der Kamera: {e}")
                return False
        return True

    def monitor_camera_access(self):
        """Überwacht den Kamera-Zugriff anderer Programme."""
        while self.camera_monitor_active:
            try:
                # Prüfe alle 2 Sekunden
                time.sleep(2)
                
                if not self.check_camera_access():
                    # Anderes Programm versucht die Kamera zu nutzen
                    self.root.after(0, self.show_camera_permission_dialog)
                    
            except Exception as e:
                print(f"Fehler in der Kameraüberwachung: {e}")
                time.sleep(5)  # Warte länger bei Fehlern

    def start_camera_monitoring(self):
        """Startet die Kameraüberwachung."""
        if not self.camera_monitor_active:
            self.camera_monitor_active = True
            self.create_camera_lock()
            self.camera_monitor_thread = threading.Thread(
                target=self.monitor_camera_access, daemon=True)
            self.camera_monitor_thread.start()
            print("Kameraüberwachung gestartet")
            
    def start_mouse_monitoring(self):
        """Startet die Mausbewegungsüberwachung."""
        if not self.mouse_monitor_active:
            self.mouse_monitor_active = True
            self.mouse_monitor_thread = threading.Thread(
                target=self.mouse_monitor_loop, daemon=True)
            self.mouse_monitor_thread.start()
            print("Mausbewegungsüberwachung gestartet")

    def stop_camera_monitoring(self):
        """Stoppt die Kameraüberwachung."""
        self.camera_monitor_active = False
        self.remove_camera_lock()
        if self.camera_permission_dialog:
            self.camera_permission_dialog.destroy()
            self.camera_permission_dialog = None
        print("Kameraüberwachung gestoppt")

    def mouse_monitor_loop(self):
        """Hauptschleife der Mausbewegungsüberwachung."""
        while self.mouse_monitor_active:
            try:
                print("Mausbewegungsüberwachung: Überprüfung läuft...")
                self.add_mouse_log("Mausbewegungsüberwachung: Überprüfung läuft...")
                data = self.collect_mouse_data(duration=3)
                if len(data) >= 20:
                    suspicious = False
                    reason = ""
                    # Modell verwenden, falls geladen
                    if self.pattern_model is not None:
                        try:
                            features = self.preprocess_mouse_data(data)
                            prediction = self.pattern_model.predict(np.array([features]))
                            if prediction[0] < 0.5:
                                suspicious = True
                                reason = "Modell: Muster nicht erkannt"
                        except Exception as e:
                            self.add_mouse_log(f"Fehler bei Modellvorhersage: {e}")
                    # Fallback/Ergänzung: Schwellenwerte
                    if self.pattern_model is None or (not suspicious):
                        # Beispielhafte Heuristik: zu wenig Bewegung
                        total_distance = sum(np.linalg.norm(np.array(data[i]) - np.array(data[i-1])) for i in range(1, len(data)))
                        if total_distance < 100:
                            suspicious = True
                            reason = "Heuristik: Zu wenig Mausbewegung"
                    if suspicious:
                        self.handle_suspicious_movement(reason)
                        break
                else:
                    self.add_mouse_log(f"Zu wenig Datenpunkte: {len(data)}")
                time.sleep(1)
            except Exception as e:
                error_msg = f"Fehler in der Mausbewegungsüberwachung: {e}"
                print("[TuxGuard] " + error_msg)
                self.add_mouse_log(error_msg)
                time.sleep(5)

    def setup_database(self):
        db_exists = os.path.exists('face_recognition.db')
        self.conn = sqlite3.connect('face_recognition.db')
        self.cursor = self.conn.cursor()
        if not db_exists:
            # Tabelle für Benutzer (nur Name und PIN)
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS users (
                                    id INTEGER PRIMARY KEY,
                                    name TEXT UNIQUE,
                                    pin_hash TEXT)''')
            # Separate Tabelle für Gesichtsbilder
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS face_encodings (
                                    id INTEGER PRIMARY KEY,
                                    user_id INTEGER,
                                    face_encoding BLOB,
                                    description TEXT,
                                    FOREIGN KEY(user_id) REFERENCES users(id))''')
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS settings
                                (setting TEXT PRIMARY KEY, value TEXT)''')
            self.conn.commit()
            messagebox.showinfo("Info", "Datenbank erstellt und verbunden")
        else:
            # Bei existierender Datenbank: Migration der alten Daten falls nötig
            try:
                self.cursor.execute("SELECT * FROM user_data LIMIT 1")
                # Wenn keine Exception, dann existiert die alte Tabelle noch
                self.migrate_old_database()
            except sqlite3.OperationalError:
                pass
            
            # Stelle sicher, dass alle Tabellen existieren
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS users (
                                    id INTEGER PRIMARY KEY,
                                    name TEXT UNIQUE,
                                    pin_hash TEXT)''')
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS face_encodings (
                                    id INTEGER PRIMARY KEY,
                                    user_id INTEGER,
                                    face_encoding BLOB,
                                    description TEXT,
                                    FOREIGN KEY(user_id) REFERENCES users(id))''')
            self.cursor.execute('''CREATE TABLE IF NOT EXISTS settings
                                (setting TEXT PRIMARY KEY, value TEXT)''')
            self.conn.commit()
            messagebox.showinfo("Info", "Mit existierender Datenbank verbunden")
    




    


    def load_image_names(self):
        self.image_listbox.delete(0, tk.END)
        self.cursor.execute("""
            SELECT u.name, COUNT(fe.id) 
            FROM users u 
            LEFT JOIN face_encodings fe ON u.id = fe.user_id 
            GROUP BY u.id, u.name
        """)
        rows = self.cursor.fetchall()
        for row in rows:
            self.image_listbox.insert(tk.END, f"{row[0]} ({row[1]} Bilder)")

    def migrate_old_database(self):
        """Migriert Daten von der alten Tabellenstruktur zur neuen"""
        self.cursor.execute("SELECT name, face_encoding, pin_hash FROM user_data")
        old_data = self.cursor.fetchall()
        
        for name, face_encoding, pin_hash in old_data:
            # Füge Benutzer hinzu
            self.cursor.execute("INSERT INTO users (name, pin_hash) VALUES (?, ?)", (name, pin_hash))
            user_id = self.cursor.lastrowid
            # Füge Gesichtsbild hinzu
            self.cursor.execute("INSERT INTO face_encodings (user_id, face_encoding, description) VALUES (?, ?, ?)",
                              (user_id, face_encoding, "Ursprüngliches Bild"))
        
        # Lösche alte Tabelle
        self.cursor.execute("DROP TABLE user_data")
        self.conn.commit()

    def capture_image_with_webcam(self):
        """Öffnet die Webcam, zeigt das Live-Bild und ermöglicht das Aufnehmen eines Fotos. Stellt sicher, dass die Kamera immer freigegeben wird."""
        import tempfile
        cam = cv2.VideoCapture(0)
        if not cam.isOpened():
            messagebox.showerror("Fehler", "Kamera konnte nicht geöffnet werden.")
            return None

        win = tk.Toplevel(self.root)
        win.title("Webcam - Foto aufnehmen")
        win.geometry("700x550")
        win.transient(self.root)
        win.grab_set()

        lmain = tk.Label(win)
        lmain.pack(pady=10)

        captured_image = {'img': None}
        camera_closed = {'done': False}

        def show_frame():
            if camera_closed['done']:
                return
            ret, frame = cam.read()
            if ret:
                rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                img = Image.fromarray(rgb)
                imgtk = ImageTk.PhotoImage(image=img)
                lmain.imgtk = imgtk
                lmain.configure(image=imgtk)
            if not camera_closed['done']:
                lmain.after(20, show_frame)
        show_frame()

        def capture():
            ret, frame = cam.read()
            if ret:
                if cam.isOpened():
                    cam.release()
                camera_closed['done'] = True
                win.destroy()
                # Speichere temporär
                tmpfile = tempfile.NamedTemporaryFile(suffix='.jpg', delete=False)
                cv2.imwrite(tmpfile.name, frame)
                captured_image['img'] = tmpfile.name
            else:
                messagebox.showerror("Fehler", "Foto konnte nicht aufgenommen werden.")

        def on_abort():
            if cam.isOpened():
                cam.release()
            camera_closed['done'] = True
            win.destroy()

        tk.Button(win, text="Foto aufnehmen", command=capture).pack(pady=10)
        tk.Button(win, text="Abbrechen", command=on_abort).pack(pady=5)
        win.protocol("WM_DELETE_WINDOW", on_abort)

        self.root.wait_window(win)
        # Sicherheitshalber Kamera freigeben, falls noch offen
        if cam.isOpened():
            cam.release()
        return captured_image['img']

    def add_new_user(self, file_path):
        """Fügt einen neuen Benutzer mit erstem Gesichtsbild hinzu"""
        name = simpledialog.askstring("Name", "Bitte geben Sie einen Namen für den neuen Benutzer ein:")
        if not name:
            messagebox.showerror("Fehler", "Name-Eingabe erforderlich.")
            return

        pin = simpledialog.askstring("PIN", "Bitte geben Sie eine PIN ein (mindestens 6 Zeichen):", show='*')
        if not pin or len(pin) < 6:
            messagebox.showerror("Fehler", "PIN muss mindestens 6 Zeichen lang sein.")
            return

        image = face_recognition.load_image_file(file_path)
        face_encodings = face_recognition.face_encodings(image)
        if not face_encodings:
            messagebox.showerror("Fehler", "Kein Gesicht im Bild erkannt.")
            return

        # Generate a random salt
        salt = os.urandom(32).hex()
        # Hash PIN with salt using stronger algorithm
        pin_hash = hashlib.pbkdf2_hmac(
            'sha256',
            pin.encode(),
            salt.encode(),
            100000  # Number of iterations
        ).hex()
        face_encoding = face_encodings[0]
        face_encoding_blob = sqlite3.Binary(np.array(face_encoding).tobytes())
        
        description = simpledialog.askstring("Beschreibung", 
                                           "Optionale Beschreibung für das Bild (z.B. 'Frontal' oder 'Mit Brille'):")
        if not description:
            description = "Hauptbild"

        # Füge neuen Benutzer hinzu
        self.cursor.execute("INSERT INTO users (name, pin_hash) VALUES (?, ?)", (name, pin_hash))
        user_id = self.cursor.lastrowid
        
        # Füge Gesichtsbild hinzu
        self.cursor.execute("INSERT INTO face_encodings (user_id, face_encoding, description) VALUES (?, ?, ?)",
                          (user_id, face_encoding_blob, description))
        
        self.conn.commit()
        self.load_image_names()
        messagebox.showinfo("Erfolg", "Neuer Benutzer mit Gesichtserkennung erfolgreich gespeichert!")

    def add_face_to_user(self, file_path, user_id):

    def start_camera(self):
        # Prüfe zuerst, ob eine Kamera verfügbar ist
        if not self.camera_available:
            messagebox.showinfo("Kamera nicht verfügbar", 
                              "Keine Kamera verfügbar.\n" +
                              "Nur die Mausbewegungserkennung ist aktiv.")
            return
        
        self.camera_active_event.set()
        
        # Starte Kameraüberwachung
        self.start_camera_monitoring()
        
        # Versuche die Kamera zu öffnen
        self.video_capture = cv2.VideoCapture(0)
        
        # Mehrere Versuche, falls die Kamera blockiert ist
        max_attempts = 3
        for attempt in range(max_attempts):
            if self.video_capture.open(0):
                print(f"Kamera erfolgreich geöffnet (Versuch {attempt + 1})")
                break
            else:
                print(f"Kamera-Versuch {attempt + 1} fehlgeschlagen")
                if attempt < max_attempts - 1:
                    time.sleep(1)  # Kurz warten vor nächstem Versuch
        else:
            # Alle Versuche fehlgeschlagen - führe Diagnose durch
            print("Alle Kamera-Versuche fehlgeschlagen - führe Diagnose durch...")
            self.diagnose_camera()
            
            print("Warnung: Kamera konnte nicht geöffnet werden, aber Überwachung läuft weiter")
            messagebox.showwarning("Kamera-Warnung", 
                                 "Kamera konnte nicht geöffnet werden.\n" +
                                 "Die Überwachung läuft trotzdem weiter.\n\n" +
                                 "Bitte führen Sie die Kamera-Diagnose aus,\n" +
                                 "um das Problem zu identifizieren.")

        self.camera_thread = threading.Thread(target=self.run_camera, daemon=True)
        self.camera_thread.start()
        self.enumerate_threads()
        if self.test_flag == False:
            self.minimize_to_tray()

    def stop_camera(self):  
        self.camera_active_event.clear()
        if hasattr(self, 'video_capture'):
            self.video_capture.release()
            cv2.destroyAllWindows()
        
        # Stoppe Kameraüberwachung
        self.stop_camera_monitoring()
        
        self.enumerate_threads()
        print("stopped")

    def run_camera(self):
        # Neue SQLite-Verbindung und Cursor im Kamera-Thread erstelle
        conn = sqlite3.connect('face_recognition.db')
        cursor = conn.cursor()

        face_not_recognized_start_time = None

        while self.camera_active_event.is_set():
            ret, frame = self.video_capture.read()
            #print(ret, frame)
            if not ret:
                break

            rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            face_locations = face_recognition.face_locations(rgb_frame)
            face_encodings = face_recognition.face_encodings(rgb_frame, face_locations)

            user_recognized = False
            for face_encoding in face_encodings:
                # Hole alle Gesichtsbilder aller Benutzer
                cursor.execute("""
                    SELECT u.name, fe.face_encoding, fe.description 
                    FROM users u 
                    JOIN face_encodings fe ON u.id = fe.user_id
                """)
                rows = cursor.fetchall()
                for row in rows:
                    db_face_encoding = np.frombuffer(row[1], dtype=np.float64)
                    matches = face_recognition.compare_faces([db_face_encoding], face_encoding)
                    if True in matches:
                        user_recognized = True
                        #print(f"Benutzer erkannt durch Bild: {row[2]}")  # Debug-Info
                        break

            if user_recognized:
                if self.test_flag == False:
                    face_not_recognized_start_time = None
                else:
                    messagebox.showinfo("Info", "Test Gesichtserkennung erfolgreich")
                    self.stop_camera()
                    self.test_flag = False                              
            else:
                if face_not_recognized_start_time is None:
                    face_not_recognized_start_time = cv2.getTickCount()
                else:
                    elapsed_time = (cv2.getTickCount() - face_not_recognized_start_time) / cv2.getTickFrequency()
                    if elapsed_time > 5:
                        if self.test_flag == False:
                            # Benutze root.after um Thread-safe zum Hauptthread zu wechseln
                            self.root.after(0, lambda: self.handle_face_not_recognized())
                            break
                        else:
                            print("Test successful")
                            self.stop_camera()
                            self.test_flag = False   
                            break
        conn.close ()
        
    def test_camera(self):

            self.test_flag = False
            self.stop_camera()
            self.close_window('test_window')
            return

        if not self.ensure_single_window('test_window'):

        self.test_flag = True

    def update_test_frame(self):
        if not self.test_flag or not hasattr(self, 'test_window'):
            return
            
        if hasattr(self, 'video_capture') and self.video_capture.isOpened():
            ret, frame = self.video_capture.read()
            if ret:
                # Konvertiere Frame für Gesichtserkennung
                rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                face_locations = face_recognition.face_locations(rgb_frame)
                face_encodings = face_recognition.face_encodings(rgb_frame, face_locations)
                
                # Prüfe Gesichtserkennung
                user_recognized = False
                if face_encodings:
                    # Hole alle Gesichtsbilder aller Benutzer
                    self.cursor.execute("""
                        SELECT u.name, fe.face_encoding, fe.description 
                        FROM users u 
                        JOIN face_encodings fe ON u.id = fe.user_id
                    """)
                    rows = self.cursor.fetchall()
                    for row in rows:
                        db_face_encoding = np.frombuffer(row[1], dtype=np.float64)
                        matches = face_recognition.compare_faces([db_face_encoding], face_encodings[0])
                        if True in matches:
                            user_recognized = True
                            self.test_status.config(text=f"Benutzer erkannt: {row[0]}")
                            break
                
                # Setze Rahmenfarbe basierend auf Erkennung
                if face_encodings:
                    if user_recognized:
                        self.test_canvas.configure(highlightbackground='green')
                        self.test_status.config(text="Benutzer erkannt!")
                    else:
                        self.test_canvas.configure(highlightbackground='red')
                        self.test_status.config(text="Gesicht erkannt, aber nicht authentifiziert")
                else:
                    self.test_canvas.configure(highlightbackground='red')
                    self.test_status.config(text="Kein Gesicht erkannt")
                
                # Konvertiere Frame für Anzeige
                frame = cv2.resize(frame, (640, 480))
                frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                photo = ImageTk.PhotoImage(image=Image.fromarray(frame))
                
                # Aktualisiere Canvas
                self.test_canvas.create_image(0, 0, image=photo, anchor=tk.NW)
                self.test_canvas.photo = photo  # Verhindere garbage collection
                
                # Plane nächste Aktualisierung
                self.test_window.after(33, self.update_test_frame)  # ~30 FPS
            else:
                self.test_status.config(text="Fehler beim Lesen der Kamera")
        else:
            self.test_status.config(text="Kamera nicht verfügbar")

    def stop_test(self):
        self.test_flag = False
        self.stop_camera()
        if hasattr(self, 'test_window'):
            self.test_window.destroy()
            delattr(self, 'test_window')

    def is_i3lock_active(self):
        for process in psutil.process_iter(['name']):
            if process.info['name'] == 'i3lock':
                return True
        return False

    def monitor_i3lock(self):
        while True:
            # Prüfen, ob i3lock läuft
            result = sp.run(["pgrep", "i3lock"], stdout=sp.PIPE, stderr=sp.PIPE)
            if result.returncode != 0:  # i3lock läuft nicht mehr
                print("i3lock entsperrt, Überwachung wird neu gestartet.")
                self.pin_dialog.destroy()
                self.start_camera()  # Überwachung neu starten
                break
            time.sleep(1)  # Überprüfung alle 1 Sekunde

    def update_counter(self):
        if not hasattr(self, 'counter') or not hasattr(self, 'pin_dialog'):
            return
            
        if self.counter > 0:
            if self.pin_dialog and self.pin_dialog.winfo_exists():
                self.counter_label.config(text=f"Verbleibende Zeit: {self.counter} Sekunden")
                self.counter -= 1
                self.root.after(1000, self.update_counter)
        else:
            if self.pin_dialog and self.pin_dialog.winfo_exists():
                self.pin_dialog.destroy()
                print("System locked due to timeout")
                # Erst i3lock starten, dann den Monitor-Thread
                sp.run(["i3lock"], restore_signals=True)
                threading.Thread(target=self.monitor_i3lock, daemon=True).start()


    def check_pin(self, reason=""):
        current_time = time.time()
        
        # Logge den PIN-Überprüfungsversuch mit detailliertem Grund
        log_message = f"PIN-Überprüfung gestartet - Grund: {reason if reason else 'keine Angabe'}"
        if hasattr(self, 'last_failed_attempts'):
            log_message += f" (Fehlversuche: {self.last_failed_attempts})"
        self.add_mouse_log(log_message)
        
        # Check if we're in a lockout period
        if hasattr(self, 'lockout_until') and current_time < self.lockout_until:
            remaining = int(self.lockout_until - current_time)
            messagebox.showerror("Fehler", f"Zu viele fehlgeschlagene Versuche. Bitte warten Sie {remaining} Sekunden.")
            return
            
        # Check cooldown period between attempts
        if hasattr(self, 'last_pin_attempt') and current_time - self.last_pin_attempt < 30:
            wait_time = int(30 - (current_time - self.last_pin_attempt))
            messagebox.showerror("Fehler", f"Bitte warten Sie {wait_time} Sekunden vor dem nächsten Versuch.")
            return
            
        pin_input = self.pin_entry.get()
        if pin_input:
            # Use constant-time comparison for PIN hash
            def constant_time_compare(a, b):
                if len(a) != len(b):
                    return False
                result = 0
                for x, y in zip(a, b):
                    result |= x ^ y
                return result == 0
                
            pin_input_hash = hashlib.sha256(pin_input.encode()).hexdigest()
            self.cursor.execute("SELECT pin_hash FROM users LIMIT 1")
            row = self.cursor.fetchone()
            
            # Update attempt timestamp
            self.last_pin_attempt = current_time
            
            if row and constant_time_compare(pin_input_hash.encode(), row[0].encode()):
                # Nach erfolgreicher PIN-Eingabe Mausbewegungen überprüfen
                is_pattern_valid = self.verify_mouse_pattern()
                if is_pattern_valid:
                    messagebox.showinfo("Erfolg", "Authentifizierung erfolgreich.")
                    if self.pin_dialog and self.pin_dialog.winfo_exists():
                        self.pin_dialog.destroy()
                    self.counter = 0  # Timer zurücksetzen
                    if self.camera_available:
                        self.start_camera()
                else:
                    messagebox.showerror("Fehler", 
                                       "Mausbewegungsmuster nicht erkannt!\n" +
                                       "Der Rechner wird in den Bereitschaftsmodus versetzt.")
                    sp.run(["i3lock"])
                    threading.Thread(target=self.monitor_i3lock, daemon=True).start()
            else:
                # Increment failed attempts
                if not hasattr(self, 'failed_pin_attempts'):
                    self.failed_pin_attempts = 0
                self.failed_pin_attempts += 1
                
                # Check if we should trigger lockout
                if self.failed_pin_attempts >= 3:
                    self.lockout_until = time.time() + 300  # 5 minute lockout
                    self.failed_pin_attempts = 0
                    messagebox.showerror("Fehler", 
                        "Zu viele fehlgeschlagene Versuche. System wird für 5 Minuten gesperrt.")
                else:
                    remaining = 3 - self.failed_pin_attempts
                    messagebox.showerror("Fehler", 
                        f"Falsche PIN. Noch {remaining} Versuche übrig. Der Rechner wird gesperrt.")
                        
                sp.run(["i3lock"])
                threading.Thread(target=self.monitor_i3lock, daemon=True).start()
        else:
            messagebox.showerror("Fehler", 
                "Keine PIN eingegeben. Der Rechner wird in den Bereitschaftsmodus versetzt.")
            sp.run(["i3lock"])
            threading.Thread(target=self.monitor_i3lock, daemon=True).start()

    # Mustererkennungsfunktionen
    def on_mouse_move(self, x, y):
        """Callback-Funktion für Mausbewegungen."""
        if self.recording:
            timestamp = time.time()
            self.mouse_data.append([x, y, timestamp, 0, 0])

    def on_mouse_click(self, x, y, button, pressed):
        """Callback-Funktion für Mausklicks."""
        if self.recording:
            timestamp = time.time()
            click_flag = 1 if pressed else 0
            self.mouse_data.append([x, y, timestamp, click_flag, 0])

    def on_mouse_scroll(self, x, y, dx, dy):
        """Callback-Funktion für Scrollen."""
        if self.recording:
            timestamp = time.time()
            self.mouse_data.append([x, y, timestamp, 0, dy])

    def collect_mouse_data(self, duration=10):
        """Sammelt Mausbewegungen für eine bestimmte Dauer."""
        try:
            print(f"Starte Mausdaten-Erfassung für {duration} Sekunden...")
            self.mouse_data = []
            self.recording = True
            
            print("Initialisiere Maus-Listener...")
            with Listener(on_move=self.on_mouse_move, 
                         on_click=self.on_mouse_click, 
                         on_scroll=self.on_mouse_scroll) as listener:
                print(f"Erfassung der Mausbewegungen für {duration} Sekunden gestartet...")
                time.sleep(duration)
                self.recording = False
                listener.stop()
                print("Listener gestoppt")
            
            print("Erfassung beendet.")
            data_array = np.array(self.mouse_data)
            print(f"Gesammelte Datenpunkte: {len(data_array)}")
            if len(data_array) > 0:
                print(f"Daten-Shape: {data_array.shape}")
                print(f"Erste 3 Datenpunkte: {data_array[:3]}")
            return data_array
        except Exception as e:
            error_msg = f"Fehler beim Sammeln der Mausdaten: {str(e)}\n\nTraceback:\n{traceback.format_exc()}"
            print("=== MAUSDATEN FEHLER ===")
            print(error_msg)
            print("========================")
            raise

    def preprocess_mouse_data(self, data):
        """Vorverarbeitung der Mausbewegungsdaten."""
        if len(data) < 2:
            raise ValueError("Nicht genügend Daten für die Verarbeitung.")
        positions = data[:, :2]
        timestamps = data[:, 2]
        clicks = data[:, 3]
        scrolls = data[:, 4]
        time_diffs = np.diff(timestamps, prepend=timestamps[0])
        features = np.hstack([positions, time_diffs.reshape(-1, 1), 
                            clicks.reshape(-1, 1), scrolls.reshape(-1, 1)])
        return features

    def build_pattern_model(self, input_shape):
        """Erstellt das neuronale Netzwerk für die Mustererkennung."""
        model = keras.models.Sequential([
            keras.layers.Input(shape=input_shape),
            keras.layers.Dense(64, activation='relu'),
            keras.layers.Dropout(0.2),
            keras.layers.Dense(32, activation='relu'),
            keras.layers.Dropout(0.2),
            keras.layers.Dense(1, activation='sigmoid')
        ])
        model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
        return model

    def start_pattern_training(self):
        """Startet das Training der Mustererkennung."""
        try:
            # GUI für das Training
            training_window = Toplevel(self.root)
            training_window.title("Mustererkennungs-Training")
            training_window.geometry("400x300")
            training_window.transient(self.root)
            
            Label(training_window, text="Training der Mausbewegungserkennung",
                  font=('Arial', 12, 'bold')).pack(pady=10)
            
            Label(training_window, text="Es werden mehrere Aufnahmen Ihrer Mausbewegungen gemacht.\n" +
                  "Bewegen Sie die Maus natürlich, wie Sie es normalerweise tun würden.",
                  wraplength=350).pack(pady=10)
            
            progress_var = tk.DoubleVar()
            progress_bar = ttk.Progressbar(training_window, variable=progress_var, maximum=100)
            progress_bar.pack(pady=10, padx=20, fill=tk.X)
            
            status_label = Label(training_window, text="Bereit zum Start")
            status_label.pack(pady=10)
            
            def collect_training_data():
                try:
                    def update_status(text):
                        print(f"Status: {text}")
                        self.add_mouse_log(f"Training Status: {text}")
                        training_window.after(0, lambda: status_label.config(text=text))
                    def update_progress(value):
                        print(f"Fortschritt: {value}%")
                        training_window.after(0, lambda: progress_var.set(value))

                    # Positive Beispiele (authentische Bewegungen)
                    update_status("Sammle positive Beispiele (normale Mausbewegungen)...")
                    positive_samples = []
                    for i in range(3):
                        update_progress((i/6) * 100)
                        print(f"Starte Aufnahme {i+1}/3 (positiv)...")
                        self.add_mouse_log(f"Training: Starte Aufnahme {i+1}/3 (positiv)")
                        data = self.collect_mouse_data(duration=5)
                        print(f"Aufnahme {i+1} abgeschlossen. Datenpunkte: {len(data)}")
                        self.add_mouse_log(f"Training: Aufnahme {i+1} abgeschlossen - {len(data)} Datenpunkte (positiv)")
                        if len(data) > 0:
                            features = self.preprocess_mouse_data(data)
                            positive_samples.append(features)
                            print(f"Features für Aufnahme {i+1}: {features.shape}")
                            self.add_mouse_log(f"Training: Features für Aufnahme {i+1}: {features.shape} (positiv)")
                        time.sleep(0.5)

                    # Negative Beispiele (falsche Bewegungen)
                    update_status("Sammle negative Beispiele (ungewöhnliche/falsche Bewegungen)...")
                    messagebox.showinfo(
                        "Negative Beispiele",
                        "Bitte führen Sie jetzt absichtlich untypische, ruckartige oder unnatürliche Mausbewegungen aus!\n\n" +
                        "Beispiel: Zickzack, Kreise, schnelle Richtungswechsel, etc.")
                    negative_samples = []
                    for i in range(3):
                        update_progress(((i+3)/6) * 100)
                        print(f"Starte Aufnahme {i+1}/3 (negativ)...")
                        self.add_mouse_log(f"Training: Starte Aufnahme {i+1}/3 (negativ)")
                        data = self.collect_mouse_data(duration=5)
                        print(f"Aufnahme {i+1} abgeschlossen. Datenpunkte: {len(data)}")
                        self.add_mouse_log(f"Training: Aufnahme {i+1} abgeschlossen - {len(data)} Datenpunkte (negativ)")
                        if len(data) > 0:
                            features = self.preprocess_mouse_data(data)
                            negative_samples.append(features)
                            print(f"Features für Aufnahme {i+1}: {features.shape}")
                            self.add_mouse_log(f"Training: Features für Aufnahme {i+1}: {features.shape} (negativ)")
                        time.sleep(0.5)

                    update_status("Generiere Trainingsmodell...")
                    update_progress(70)

                    # Trainingsdaten und Labels
                    if not positive_samples or not negative_samples:
                        raise ValueError("Nicht genügend Trainingsdaten gesammelt (positiv/negativ)")
                    X_pos = np.vstack(positive_samples)
                    y_pos = np.ones(len(X_pos))
                    X_neg = np.vstack(negative_samples)
                    y_neg = np.zeros(len(X_neg))
                    X_train = np.vstack([X_pos, X_neg])
                    y_train = np.concatenate([y_pos, y_neg])
                    print(f"Trainingsdaten: X_train Shape: {X_train.shape}, y_train Shape: {y_train.shape}")
                    self.add_mouse_log(f"Training: Trainingsdaten vorbereitet - X: {X_train.shape}, y: {y_train.shape}")

                    # Shuffle
                    idx = np.arange(len(X_train))
                    np.random.shuffle(idx)
                    X_train = X_train[idx]
                    y_train = y_train[idx]

                    # Modell
                    # Debug: Label-Verteilung und Beispielwerte
                    unique, counts = np.unique(y_train, return_counts=True)
                    print(f"Label-Verteilung: {dict(zip(unique, counts))}")
                    print(f"Erste 5 Labels: {y_train[:5]}")
                    print(f"Erste 5 Trainingsdaten: {X_train[:5]}")
                    # Debug: Modellvorhersagen für die ersten Trainingsdaten
                    if self.pattern_model:
                        preds = self.pattern_model.predict(X_train[:5])
                        print("Debug: Vorhersagen für erste 5 Trainingsdaten:")
                        for i, pred in enumerate(preds):
                            print(f"Sample {i}: Label={y_train[i]}, Confidence={pred[0] if isinstance(pred, (list, np.ndarray)) else pred}")
                    input_shape = (X_train.shape[1],)
                    print(f"Input Shape: {input_shape}")
                    self.add_mouse_log(f"Training: Input Shape: {input_shape}")
                    self.pattern_model = self.build_pattern_model(input_shape)

                    update_status("Trainiere Modell...")
                    update_progress(80)
                    print("Starte Modell-Training...")
                    self.add_mouse_log("Training: Starte Modell-Training...")
                    self.pattern_model.fit(X_train, y_train, epochs=20, batch_size=32, verbose=1)
                    print("Modell-Training abgeschlossen")
                    self.add_mouse_log("Training: Modell-Training abgeschlossen")

                    # Speichere das Modell
                    update_status("Speichere Modell...")
                    update_progress(90)
                    model_dir = Path("models")
                    model_dir.mkdir(exist_ok=True)
                    model_path = model_dir / "mouse_pattern_model.keras"
                    self.pattern_model.save(str(model_path))
                    print(f"Modell gespeichert in: {model_path}")
                    self.add_mouse_log(f"Training: Modell gespeichert in: {model_path}")

                    update_progress(100)
                    update_status("Training abgeschlossen!")
                    self.add_mouse_log("Training: Training erfolgreich abgeschlossen!")

                    training_window.after(0, lambda: messagebox.showinfo(
                        "Erfolg",
                        "Das Training wurde erfolgreich abgeschlossen!\n" +
                        "Die Mustererkennung ist jetzt aktiv."))
                    training_window.after(100, training_window.destroy)
                except Exception as e:
                    error_msg = f"Fehler beim Training: {str(e)}\n\nTraceback:\n{traceback.format_exc()}"
                    print("=== TRAINING FEHLER ===")
                    print(error_msg)
                    print("=======================")
                    training_window.after(0, lambda: messagebox.showerror(
                        "Fehler",
                        f"Fehler beim Training: {str(e)}"))
                    training_window.after(100, training_window.destroy)
            ttk.Button(training_window, text="Training starten", 
                      command=lambda: threading.Thread(target=collect_training_data, 
                                                    daemon=True).start()).pack(pady=10)
        except Exception as e:
            error_msg = f"Fehler beim Erstellen des Training-Fensters: {str(e)}\n\nTraceback:\n{traceback.format_exc()}"
            print("=== FENSTER FEHLER ===")
            print(error_msg)
            print("======================")
            messagebox.showerror("Fehler", f"Fehler beim Erstellen des Training-Fensters: {str(e)}")

    def verify_mouse_pattern(self, duration=5):
        """Überprüft das aktuelle Mausbewegungsmuster."""
        # Modell ist beim Start geladen, kein erneutes Laden hier
        if self.pattern_model is None:
            self.add_mouse_log("Kein Mustererkennungsmodell geladen, Zugriff erlaubt.")
            return True
        data = self.collect_mouse_data(duration=duration)
        if len(data) < 2:
            self.add_mouse_log("Zu wenig Mausdaten für Mustererkennung, Zugriff erlaubt.")
            return True
        try:
            features = self.preprocess_mouse_data(data)
            predictions = self.pattern_model.predict(features, verbose=0)
            avg_pred = predictions.mean()
            logmsg = f"Verifikation Modell-Score: {avg_pred:.3f} (Schwelle: 0.5)"
            print("[TuxGuard] " + logmsg)
            self.add_mouse_log(logmsg)
            is_auth = avg_pred > 0.5
            # --- Kontinuierliches Lernen: Bei erfolgreicher Authentifizierung Mausdaten speichern ---
            if is_auth and self.adaptive_learning_enabled:
                # Nur positive Beispiele (authentische Nutzung)
                self.adaptive_positive_samples.append(features)
                self.adaptive_auth_success_count += 1
                self.add_mouse_log(f"Adaptives Lernen: Erfolg #{self.adaptive_auth_success_count}, neue Daten gespeichert.")
                # Nach jedem N-ten Erfolg: Modell nachtrainieren
                if self.adaptive_auth_success_count % self.adaptive_retrain_interval == 0:
                    self.add_mouse_log("Adaptives Lernen: Starte automatisches Nachtraining...")
                    self.retrain_pattern_model_adaptive()
            return is_auth
        except Exception as e:
            errmsg = f"Fehler bei Modell-Verifikation: {e}"
            print("[TuxGuard] " + errmsg)
            self.add_mouse_log(errmsg)
            return True  # Im Fehlerfall nicht sperren
    def retrain_pattern_model_adaptive(self):
        """Trainiert das Modell mit den gesammelten neuen positiven Beispielen nach."""
        if not self.adaptive_positive_samples:
            self.add_mouse_log("Adaptives Lernen: Keine neuen Daten zum Nachtrainieren.")
            return
        try:
            # Lade negatives Ursprungs-Set (aus Datei, falls vorhanden)
            model_dir = Path("models")
            neg_path = model_dir / "mouse_pattern_negatives.npy"
            if neg_path.exists():
                X_neg = np.load(neg_path)
                y_neg = np.zeros(len(X_neg))
            else:
                self.add_mouse_log("Warnung: Keine negativen Beispiele für Nachtraining gefunden. Nur positives Nachtraining.")
                X_neg = np.empty((0, self.adaptive_positive_samples[0].shape[1]))
                y_neg = np.zeros(0)
            # Neue positive Beispiele
            X_pos = np.vstack(self.adaptive_positive_samples)
            y_pos = np.ones(len(X_pos))
            # Kombiniere mit bisherigen Trainingsdaten (optional: lade positives Ursprungs-Set)
            # Hier: Nur neue positive + negatives Ursprungs-Set
            X_train = np.vstack([X_pos, X_neg]) if len(X_neg) > 0 else X_pos
            y_train = np.concatenate([y_pos, y_neg]) if len(y_neg) > 0 else y_pos
            # Shuffle
            idx = np.arange(len(X_train))
            np.random.shuffle(idx)
            X_train = X_train[idx]
            y_train = y_train[idx]
            # Modellstruktur beibehalten
            input_shape = (X_train.shape[1],)
            self.pattern_model = self.build_pattern_model(input_shape)
            self.pattern_model.fit(X_train, y_train, epochs=10, batch_size=32, verbose=0)
            # Speichere Modell
            model_path = model_dir / "mouse_pattern_model.keras"
            self.pattern_model.save(str(model_path))
            self.add_mouse_log(f"Adaptives Lernen: Modell nachtrainiert und gespeichert ({len(X_pos)} neue positive Beispiele)")
            # Leere den Zwischenspeicher
            self.adaptive_positive_samples.clear()
        except Exception as e:
            self.add_mouse_log(f"Fehler beim adaptiven Nachtraining: {e}")
        except Exception as e:
            errmsg = f"Fehler bei Modell-Verifikation: {e}"
            print("[TuxGuard] " + errmsg)
            self.add_mouse_log(errmsg)
            return True  # Im Fehlerfall nicht sperren

    def diagnose_camera(self):
        """Führt eine umfassende Kamera-Diagnose durch."""
        print("=== KAMERA-DIAGNOSE ===")
        
        # 1. Prüfe verfügbare Video-Devices
        print("1. Verfügbare Video-Devices:")
        for i in range(10):  # Prüfe /dev/video0 bis /dev/video9
            device_path = f"/dev/video{i}"
            if os.path.exists(device_path):
                print(f"   ✓ {device_path} existiert")
                try:
                    # Prüfe Berechtigungen
                    stat_info = os.stat(device_path)
                    print(f"      Berechtigungen: {oct(stat_info.st_mode)[-3:]}")
                    print(f"      Besitzer: {stat_info.st_uid}")
                except Exception as e:
                    print(f"      Fehler beim Prüfen der Berechtigungen: {e}")
            else:
                print(f"   ✗ {device_path} existiert nicht")
        
        # 2. Prüfe OpenCV-Kamera-Zugriff
        print("\n2. OpenCV-Kamera-Test:")
        for i in range(5):  # Teste Kamera 0-4
            try:
                cap = cv2.VideoCapture(i)
                if cap.isOpened():
                    ret, frame = cap.read()
                    if ret:
                        print(f"   ✓ Kamera {i}: Funktioniert (Frame: {frame.shape})")
                    else:
                        print(f"   ⚠ Kamera {i}: Geöffnet, aber kein Frame lesbar")
                    cap.release()
                else:
                    print(f"   ✗ Kamera {i}: Kann nicht geöffnet werden")
            except Exception as e:
                print(f"   ✗ Kamera {i}: Fehler - {e}")
        
        # 3. Prüfe System-Informationen
        print("\n3. System-Informationen:")
        try:
            # Prüfe v4l2-utils (falls installiert)
            result = sp.run(['v4l2-ctl', '--list-devices'], 
                          capture_output=True, text=True, timeout=5)
            if result.returncode == 0:
                print("   v4l2-ctl verfügbar:")
                for line in result.stdout.split('\n')[:10]:  # Erste 10 Zeilen
                    if line.strip():
                        print(f"      {line}")
            else:
                print("   v4l2-ctl nicht verfügbar")
        except (sp.TimeoutExpired, FileNotFoundError):
            print("   v4l2-ctl nicht installiert oder nicht verfügbar")
        
        # 4. Prüfe laufende Prozesse
        print("\n4. Prozesse mit Kamera-Zugriff:")
        try:
            result = sp.run(['lsof', '/dev/video*'], 
                          capture_output=True, text=True, timeout=5)
            if result.returncode == 0 and result.stdout.strip():
                print("   Prozesse mit Kamera-Zugriff:")
                for line in result.stdout.split('\n'):
                    if line.strip():
                        print(f"      {line}")
            else:
                print("   Keine Prozesse mit Kamera-Zugriff gefunden")
        except (sp.TimeoutExpired, FileNotFoundError):
            print("   lsof nicht verfügbar")
        
        # 5. Prüfe Kernel-Module
        print("\n5. Kamera-Kernel-Module:")
        try:
            result = sp.run(['lsmod'], capture_output=True, text=True, timeout=5)
            if result.returncode == 0:
                camera_modules = ['uvcvideo', 'videodev', 'v4l2_common', 'v4l2loopback']
                for module in camera_modules:
                    if module in result.stdout:
                        print(f"   ✓ {module} geladen")
                    else:
                        print(f"   ✗ {module} nicht geladen")
        except (sp.TimeoutExpired, FileNotFoundError):
            print("   lsmod nicht verfügbar")
        
        print("=== ENDE KAMERA-DIAGNOSE ===\n")

    def check_camera_availability(self):
        """Prüft, ob eine Kamera verfügbar ist."""
        try:
            # Prüfe Video-Devices
            for i in range(5):  # Prüfe /dev/video0 bis /dev/video4
                device_path = f"/dev/video{i}"
                if os.path.exists(device_path):
                    # Versuche die Kamera zu öffnen
                    cap = cv2.VideoCapture(i)
                    if cap.isOpened():
                        ret, frame = cap.read()
                        cap.release()
                        if ret:
                            print(f"Kamera gefunden: {device_path}")
                            return True
            return False
        except Exception as e:
            print(f"Fehler beim Prüfen der Kamera-Verfügbarkeit: {e}")
            return False

    def start_monitoring(self):
        """Startet die Überwachung basierend auf verfügbaren Komponenten."""
        if not self.monitoring_active:
            # Prüfe, ob ein trainiertes Mausbewegungsmodell vorhanden ist
            model_path = Path("models/mouse_pattern_model.keras")
            if not model_path.exists():
                messagebox.showwarning("Kein trainiertes Modell", 
                                     "Bitte trainieren Sie zuerst die Mausbewegungserkennung,\n" +
                                     "bevor Sie die Überwachung starten.")
                return
            
            self.monitoring_active = True
            self.start_monitoring_button.config(state='disabled')
            self.stop_monitoring_button.config(state='normal')
            
            # Lade das Mausbewegungsmodell
            try:
                self.pattern_model = keras.models.load_model(str(model_path))
                print("Mausbewegungsmodell geladen")
                self.add_mouse_log("Mausbewegungsmodell geladen")
            except Exception as e:
                print(f"Fehler beim Laden des Modells: {e}")
                self.add_mouse_log(f"Fehler beim Laden des Modells: {e}")
                messagebox.showerror("Fehler", "Konnte Mausbewegungsmodell nicht laden.")
                self.stop_monitoring()
                return
            
            # Starte Überwachung basierend auf verfügbaren Komponenten
            if self.camera_available:
                print("Starte Überwachung: Kamera + Mausbewegungserkennung")
                self.add_mouse_log("Starte Überwachung: Kamera + Mausbewegungserkennung")
                self.status_label.config(text="✓ Überwachung aktiv (Kamera + Mausbewegung)", fg="green")
                self.start_camera_monitoring()
                self.start_camera()
            else:
                print("Starte Überwachung: Nur Mausbewegungserkennung")
                self.add_mouse_log("Starte Überwachung: Nur Mausbewegungserkennung")
                self.status_label.config(text="✓ Überwachung aktiv (Nur Mausbewegung)", fg="green")
                self.start_mouse_monitoring_only()
            
            # Minimiere zu Tray
            self.minimize_to_tray()

    def stop_monitoring(self):
        """Stoppt die Überwachung."""
        self.monitoring_active = False
        self.start_monitoring_button.config(state='normal')
        self.stop_monitoring_button.config(state='disabled')
        
        # Stoppe alle Überwachungskomponenten
        if self.camera_available:
            self.stop_camera()
            self.stop_camera_monitoring()
            self.add_mouse_log("Überwachung gestoppt (Kamera + Mausbewegung)")
        else:
            self.stop_mouse_monitoring_only()
            self.add_mouse_log("Überwachung gestoppt (Nur Mausbewegung)")
        
        print("Überwachung gestoppt")
        if self.camera_available:
            self.status_label.config(text="✓ Kamera verfügbar\n✓ Mausbewegungserkennung verfügbar", fg="green")
        else:
            self.status_label.config(text="✗ Kamera nicht verfügbar\n✓ Mausbewegungserkennung verfügbar", fg="orange")

    def retrain_pattern_model_adaptive(self):
        """Trainiert das Modell mit den gesammelten neuen positiven Beispielen nach."""
        if not self.adaptive_positive_samples:
            self.add_mouse_log("Adaptives Lernen: Keine neuen Daten zum Nachtrainieren.")
            return
        try:
            # Lade negatives Ursprungs-Set (aus Datei, falls vorhanden)
            model_dir = Path("models")
            neg_path = model_dir / "mouse_pattern_negatives.npy"
            if neg_path.exists():
                X_neg = np.load(neg_path)
                y_neg = np.zeros(len(X_neg))
            else:
                self.add_mouse_log("Warnung: Keine negativen Beispiele für Nachtraining gefunden. Nur positives Nachtraining.")
                X_neg = np.empty((0, self.adaptive_positive_samples[0].shape[1]))
                y_neg = np.zeros(0)
            # Neue positive Beispiele
            X_pos = np.vstack(self.adaptive_positive_samples)
            y_pos = np.ones(len(X_pos))
            # Kombiniere mit bisherigen Trainingsdaten (optional: lade positives Ursprungs-Set)
            # Hier: Nur neue positive + negatives Ursprungs-Set
            X_train = np.vstack([X_pos, X_neg]) if len(X_neg) > 0 else X_pos
            y_train = np.concatenate([y_pos, y_neg]) if len(y_neg) > 0 else y_pos
            # Shuffle
            idx = np.arange(len(X_train))
            np.random.shuffle(idx)
            X_train = X_train[idx]
            y_train = y_train[idx]
            # Modellstruktur beibehalten
            input_shape = (X_train.shape[1],)
            self.pattern_model = self.build_pattern_model(input_shape)
            self.pattern_model.fit(X_train, y_train, epochs=10, batch_size=32, verbose=0)
            # Speichere Modell
            model_path = model_dir / "mouse_pattern_model.keras"
            self.pattern_model.save(str(model_path))
            self.add_mouse_log(f"Adaptives Lernen: Modell nachtrainiert und gespeichert ({len(X_pos)} neue positive Beispiele)")
            # Leere den Zwischenspeicher
            self.adaptive_positive_samples.clear()
        except Exception as e:
            self.add_mouse_log(f"Fehler beim adaptiven Nachtraining: {e}")
        # Beispielhafter Log-Export (korrektes lines-Handling)
        # ...existing code...
        # lines = ... (z.B. lines = self.mouse_logs_text.get('1.0', 'end').splitlines())
        # if len(lines) > 1000:
            # Entferne die ersten 100 Zeilen
